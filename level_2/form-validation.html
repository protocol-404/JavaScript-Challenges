Valider un formulaire avec plusieurs champs

Retour

Description

Ajoutez une validation pour les champs comme e-mail et mot de passe.

Solution


const inputField = document.getElementById('inputField');
const inputField2 = document.getElementById('inputField2');
const inputField3 = document.getElementById('inputField3');
const inputField4 = document.getElementById('inputField4');
const inputField5 = document.getElementById('inputField5');
const validateButton = document.getElementById('validateButton');

validateButton.addEventListener('click', () => {
    const name = inputField.value;
    const prenom = inputField2.value;
    const age = inputField3.value;
    const email = inputField4.value;
    const password = inputField5.value;

    const nameRegex = /^[a-zA-Z\s]{5,}$/;
    const prenomRegex = /^[a-zA-Z\s]{5,}$/;
    const ageRegex = /^[0-9]{2}$/;
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    const passwordRegex = /^(?=.*[A-Za-z])(?=.*\d)[A-Za-z\d]{8,}$/;

    if (nameRegex.test(name) && prenomRegex.test(prenom) && ageRegex.test(age) && emailRegex.test(email) && passwordRegex.test(password)) {
        alert('Formulaire validé');
    } else {
        alert('Formulaire non validé');
    }
});
