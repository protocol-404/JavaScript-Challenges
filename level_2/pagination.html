Système de pagination

Retour

Description

Créez une liste paginée pour gérer un grand nombre d'éléments.

Solution


const itemsPerPage = 5;
let currentPage = 1;

function displayItems(page) {
    const items = document.querySelectorAll('.page-item');
    const startIndex = (page - 1) * itemsPerPage;
    const endIndex = startIndex + itemsPerPage;
    
    items.forEach((item, index) => {
        if (index >= startIndex && index < endIndex) {
            item.classList.remove('hidden');
        } else {
            item.classList.add('hidden');
        }
    });
}

function updatePagination() {
    const items = document.querySelectorAll('.page-item');
    const totalPages = Math.ceil(items.length / itemsPerPage);
    const paginationContainer = document.getElementById('pagination');
    paginationContainer.innerHTML = '';
    
    // Bouton précédent
    const prevButton = createPageButton('Précédent', () => {
        if (currentPage > 1) {
            currentPage--;
            updateDisplay();
        }
    });
    prevButton.classList.toggle('opacity-50', currentPage === 1);
    paginationContainer.appendChild(prevButton);
    
    // Boutons des pages
    for (let i = 1; i <= totalPages; i++) {
        const pageButton = createPageButton(i, () => {
            currentPage = i;
            updateDisplay();
        });
        if (i === currentPage) {
            pageButton.classList.add('active', 'bg-blue-500', 'text-white');
        }
        paginationContainer.appendChild(pageButton);
    }
    
    // Bouton suivant
    const nextButton = createPageButton('Suivant', () => {
        if (currentPage < totalPages) {
            currentPage++;
            updateDisplay();
        }
    });
    nextButton.classList.toggle('opacity-50', currentPage === totalPages);
    paginationContainer.appendChild(nextButton);
}

function createPageButton(text, onClick) {
    const button = document.createElement('button');
    button.textContent = text;
    button.className = 'page-button mx-1 px-3 py-1 rounded hover:bg-blue-100';
    button.addEventListener('click', onClick);
    return button;
}

function updateDisplay() {
    displayItems(currentPage);
    updatePagination();
}
