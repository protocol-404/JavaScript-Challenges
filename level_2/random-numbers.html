Description

Display random numbers in a dynamic table.

Solution


const numbersTable = document.getElementById('numbersTable');
const generateBtn = document.getElementById('generateBtn');
const clearBtn = document.getElementById('clearBtn');

function generateRandomNumber(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

function addNumber() {
    const number = generateRandomNumber(1, 100);
    const row = numbersTable.insertRow();
    const cell1 = row.insertCell(0);
    const cell2 = row.insertCell(1);
    
    cell1.textContent = numbersTable.rows.length - 1;
    cell2.textContent = number;
    
    cell1.className = 'number-cell border px-4 py-2';
    cell2.className = 'number-cell border px-4 py-2';
    
    numbersTable.parentElement.scrollTop = numbersTable.parentElement.scrollHeight;
}

generateBtn.addEventListener('click', addNumber);

clearBtn.addEventListener('click', () => {
    while(numbersTable.rows.length > 1) {
        numbersTable.deleteRow(1);
    }
});
