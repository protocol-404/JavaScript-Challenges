Implémenter un mode sombre et un mode clair

Retour

Description

Ajoutez un bouton pour basculer entre les modes sombre et clair.

Solution


const ThemeManager = {
    isDark: false,
    
    init() {
        const savedTheme = localStorage.getItem('theme');
        this.isDark = savedTheme === 'dark';
        this.applyTheme();
    },
    
    toggleTheme() {
        this.isDark = !this.isDark;
        this.applyTheme();
        this.saveTheme();
    },
    
    applyTheme() {
        const demoContent = document.querySelector('.demo-content');
        const toggleSwitch = document.querySelector('.toggle-switch');
        
        if (this.isDark) {
            demoContent.classList.add('dark');
            toggleSwitch.classList.add('dark');
        } else {
            demoContent.classList.remove('dark');
            toggleSwitch.classList.remove('dark');
        }
    },
    
    saveTheme() {
        localStorage.setItem('theme', this.isDark ? 'dark' : 'light');
    }
};

const UI = {
    demoContent: document.querySelector('.demo-content'),
    toggleSwitch: document.querySelector('.toggle-switch'),
    
    init() {
        ThemeManager.init();
        this.setupEventListeners();
    },
    
    setupEventListeners() {
        this.toggleSwitch.addEventListener('click', () => {
            ThemeManager.toggleTheme();
        });
    }
};
