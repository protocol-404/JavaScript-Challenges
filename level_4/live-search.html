Description

Créez une barre de recherche qui interagit en direct avec une API.

Solution


const SearchManager = {
    searchTimeout: null,
    currentRequest: null,
    minChars: 2,
    debounceTime: 300,
    
    init() {
        this.setupEventListeners();
        this.updateStats(0, 0);
    },
    
    setupEventListeners() {
        const searchInput = document.getElementById('searchInput');
        searchInput.addEventListener('input', (e) => this.handleSearch(e.target.value));
        searchInput.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                this.clearSearch();
            }
        });
        
        document.getElementById('clearButton').addEventListener('click', () => {
            this.clearSearch();
        });
    },
    
    async handleSearch(query) {
        // Update clear button visibility
        document.getElementById('clearButton').style.display = query ? 'block' : 'none';
        
        if (query.length < this.minChars) {
            this.clearResults();
            this.updateStats(0, 0);
            return;
        }
        
        // Cancel previous timeout
        if (this.searchTimeout) {
            clearTimeout(this.searchTimeout);
        }
        
        // Cancel previous request
        if (this.currentRequest) {
            this.currentRequest.abort();
        }
        
        // Show loading state
        this.showLoading();
        
        // Debounce search
        this.searchTimeout = setTimeout(async () => {
            try {
                await this.performSearch(query);
            } catch (error) {
                if (error.name === 'AbortError') return;
                this.showError('Erreur lors de la recherche');
            }
        }, this.debounceTime);
    },
    
    async performSearch(query) {
        // Create abort controller for this request
        const controller = new AbortController();
        this.currentRequest = controller;
        
        try {
            const response = await fetch(
                `https://jsonplaceholder.typicode.com/posts?q=${encodeURIComponent(query)}`,
                { signal: controller.signal }
            );
            
            if (!response.ok) throw new Error('Erreur réseau');
            
            const data = await response.json();
            this.displayResults(data, query);
            this.updateStats(data.length, Date.now());
        } finally {
            this.hideLoading();
        }
    },
    
    displayResults(results, query) {
        const container = document.getElementById('resultsContainer');
        container.textContent = '';
        
        if (results.length === 0) {
            const noResults = document.createElement('div');
            noResults.className = 'text-gray-500 text-center py-8';
            noResults.textContent = 'Aucun résultat trouvé';
            container.appendChild(noResults);
            return;
        }
        
        results.forEach(result => {
            const item = document.createElement('div');
            item.className = 'result-item bg-white p-4 rounded-lg shadow mb-4';
            
            const title = document.createElement('h3');
            title.className = 'font-semibold text-gray-800 mb-2';
            title.appendChild(this.highlightText(result.title, query));
            
            const body = document.createElement('p');
            body.className = 'text-gray-600 text-sm';
            body.appendChild(this.highlightText(result.body, query));
            
            item.appendChild(title);
            item.appendChild(body);
            container.appendChild(item);
        });
    },
    
    highlightText(text, query) {
        const fragment = document.createDocumentFragment();
        const regex = new RegExp(`(${query})`, 'gi');
        let lastIndex = 0;
        
        text.replace(regex, (match, p1, offset) => {
            // Add text before match
            if (offset > lastIndex) {
                fragment.appendChild(document.createTextNode(text.slice(lastIndex, offset)));
            }
            
            // Add highlighted match
            const highlight = document.createElement('span');
            highlight.className = 'highlight';
            highlight.textContent = match;
            fragment.appendChild(highlight);
            
            lastIndex = offset + match.length;
        });
        
        // Add remaining text
        if (lastIndex < text.length) {
            fragment.appendChild(document.createTextNode(text.slice(lastIndex)));
        }
        
        return fragment;
    },
    
    clearSearch() {
        const searchInput = document.getElementById('searchInput');
        searchInput.value = '';
        searchInput.focus();
        this.clearResults();
        this.updateStats(0, 0);
        document.getElementById('clearButton').style.display = 'none';
    },
    
    clearResults() {
        const container = document.getElementById('resultsContainer');
        container.textContent = '';
    },
    
    showLoading() {
        document.getElementById('loadingSpinner').classList.remove('hidden');
    },
    
    hideLoading() {
        document.getElementById('loadingSpinner').classList.add('hidden');
    },
    
    showError(message) {
        const error = document.getElementById('errorMessage');
        error.textContent = message;
        error.classList.remove('hidden');
        setTimeout(() => error.classList.add('hidden'), 3000);
    },
    
    updateStats(resultCount, timestamp) {
        const stats = document.getElementById('searchStats');
        if (resultCount === 0) {
            stats.textContent = 'Entrez au moins 2 caractères pour lancer la recherche';
            return;
        }
        
        const time = new Date(timestamp).toLocaleTimeString();
        stats.textContent = `${resultCount} résultat${resultCount > 1 ? 's' : ''} trouvé${resultCount > 1 ? 's' : ''} à ${time}`;
    }
};
